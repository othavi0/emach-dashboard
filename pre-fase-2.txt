  -- prevent_category_cycle: BEFORE INSERT/UPDATE em category
  CREATE OR REPLACE FUNCTION prevent_category_cycle() RETURNS trigger AS $$
  DECLARE
    cycle_found boolean;
  BEGIN
    IF NEW.parent_id IS NULL THEN
      NEW.depth := 0;
      NEW.path := '/' || NEW.slug;
      RETURN NEW;
    END IF;

    -- recursive CTE detecta ciclo
    WITH RECURSIVE ancestors AS (
      SELECT id, parent_id, 1 AS hops FROM category WHERE id = NEW.parent_id
      UNION ALL
      SELECT c.id, c.parent_id, a.hops + 1
      FROM category c JOIN ancestors a ON c.id = a.parent_id
      WHERE a.hops < 10
    )
    SELECT EXISTS(SELECT 1 FROM ancestors WHERE id = NEW.id) INTO cycle_found;

    IF cycle_found THEN
      RAISE EXCEPTION 'category cycle detected for id %', NEW.id;
    END IF;

    -- recalcula path + depth
    SELECT path || '/' || NEW.slug, depth + 1
    INTO NEW.path, NEW.depth
    FROM category WHERE id = NEW.parent_id;

    RETURN NEW;
  END $$ LANGUAGE plpgsql;

  CREATE TRIGGER trg_prevent_category_cycle
  BEFORE INSERT OR UPDATE OF parent_id, slug ON category
  FOR EACH ROW EXECUTE FUNCTION prevent_category_cycle();
